    def load_competencies(self):
        """–ó–∞–≥—Ä—É–∂–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –∫–æ–º–ø–µ—Ç–µ–Ω—Ü–∏–π"""
        competencies = self.db.get_competencies()

        for i, (comp_id, name, category) in enumerate(competencies[:3]):  # –º–∞–∫—Å–∏–º—É–º 3
            var = tk.BooleanVar()
            level_var = tk.IntVar(value=3)

            frame = ttk.Frame(self.competencies_frame)
            frame.pack(anchor='w', pady=2)

            chk = ttk.Checkbutton(frame, text=f"{name} ({category})", variable=var)
            chk.pack(side='left')

            ttk.Label(frame, text="–£—Ä–æ–≤–µ–Ω—å:").pack(side='left', padx=(10, 5))
            level_combo = ttk.Combobox(frame, textvariable=level_var,
                                       values=[1, 2, 3, 4, 5], width=5, state="readonly")
            level_combo.pack(side='left')

            self.competency_vars.append((comp_id, var))
            self.level_vars.append(level_var)
            self.competency_widgets.append((frame, chk, level_combo))

    def update_keywords_suggestions(self):

        keywords = self.db.get_all_keywords()
        self.keywords_listbox.delete(0, tk.END)
        for keyword in keywords:
            self.keywords_listbox.insert(tk.END, keyword)

    def add_entry(self):
        """–î–æ–±–∞–≤–ª—è–µ—Ç –Ω–æ–≤—É—é –∑–∞–ø–∏—Å—å"""
        # –ü–æ–ª—É—á–∞–µ–º –¥–∞–Ω–Ω—ã–µ –∏–∑ —Ñ–æ—Ä–º—ã
        title = self.title_entry.get().strip()
        entry_type = self.type_var.get()
        date = self.date_entry.get().strip()
        description = self.desc_text.get("1.0", tk.END).strip()
        authors = self.authors_entry.get().strip()
        keywords = [k.strip() for k in self.keywords_entry.get().split(',') if k.strip()][:5]

        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–µ –ø–æ–ª—è
        if not title or not entry_type or not date:
            messagebox.showerror("–û—à–∏–±–∫–∞", "–ó–∞–ø–æ–ª–Ω–∏—Ç–µ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–µ –ø–æ–ª—è: –ù–∞–∑–≤–∞–Ω–∏–µ, –¢–∏–ø, –î–∞—Ç–∞")
            return

        # –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ñ–æ—Ä–º–∞—Ç –¥–∞—Ç—ã
        try:
            datetime.strptime(date, "%Y-%m-%d")
        except ValueError:
            messagebox.showerror("–û—à–∏–±–∫–∞", "–ù–µ–ø—Ä–∞–≤–∏–ª—å–Ω—ã–π —Ñ–æ—Ä–º–∞—Ç –¥–∞—Ç—ã. –ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ –ì–ì–ì–ì-–ú–ú-–î–î")
            return

        # –ü–æ–ª—É—á–∞–µ–º –≤—ã–±—Ä–∞–Ω–Ω—ã–µ –∫–æ–º–ø–µ—Ç–µ–Ω—Ü–∏–∏
        selected_competencies = []
        for (comp_id, var), level_var in zip(self.competency_vars, self.level_vars):
            if var.get():
                selected_competencies.append((comp_id, level_var.get()))

        if len(selected_competencies) > 3:
            messagebox.showerror("–û—à–∏–±–∫–∞", "–ú–æ–∂–Ω–æ –≤—ã–±—Ä–∞—Ç—å –Ω–µ –±–æ–ª–µ–µ 3 –∫–æ–º–ø–µ—Ç–µ–Ω—Ü–∏–π")
            return

        if len(selected_competencies) == 0:
            messagebox.showerror("–û—à–∏–±–∫–∞", "–í—ã–±–µ—Ä–∏—Ç–µ —Ö–æ—Ç—è –±—ã –æ–¥–Ω—É –∫–æ–º–ø–µ—Ç–µ–Ω—Ü–∏—é")
            return

        # –î–æ–±–∞–≤–ª—è–µ–º –∑–∞–ø–∏—Å—å –≤ –ë–î
        try:
            entry_id = self.db.add_entry(title, entry_type, date, description,
                                         authors, keywords, selected_competencies)

            # –ü—Ä–æ–≤–µ—Ä—è–µ–º –¥–æ—Å—Ç–∏–∂–µ–Ω–∏—è
            achievements = self.db.check_achievements({
                'title': title,
                'type': entry_type,
                'authors': authors,
                'description': description
            })

            if achievements:
                messagebox.showinfo("–ü–æ–∑–¥—Ä–∞–≤–ª—è–µ–º!",
                                    f"–ó–∞–ø–∏—Å—å –¥–æ–±–∞–≤–ª–µ–Ω–∞!\n\n–ü–æ–ª—É—á–µ–Ω—ã –¥–æ—Å—Ç–∏–∂–µ–Ω–∏—è:\n" +
                                    "\n".join([f"‚Ä¢ {name}" for name, _ in achievements]))
            else:
                messagebox.showinfo("–£—Å–ø–µ—Ö", "–ó–∞–ø–∏—Å—å —É—Å–ø–µ—à–Ω–æ –¥–æ–±–∞–≤–ª–µ–Ω–∞!")

            # –û—á–∏—â–∞–µ–º —Ñ–æ—Ä–º—É
            self.clear_form()

            # –û–±–Ω–æ–≤–ª—è–µ–º –¥—Ä—É–≥–∏–µ –≤–∫–ª–∞–¥–∫–∏
            self.update_research_map()
            self.update_competencies_dashboard()
            self.update_goals_display()
            self.update_achievements_display()
            self.update_keywords_suggestions()

        except Exception as e:
            messagebox.showerror("–û—à–∏–±–∫–∞", f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –¥–æ–±–∞–≤–ª–µ–Ω–∏–∏ –∑–∞–ø–∏—Å–∏: {str(e)}")

    def clear_form(self):
        """–û—á–∏—â–∞–µ—Ç —Ñ–æ—Ä–º—É –≤–≤–æ–¥–∞"""
        self.title_entry.delete(0, tk.END)
        self.type_combo.current(0)
        self.date_entry.delete(0, tk.END)
        self.date_entry.insert(0, datetime.now().strftime("%Y-%m-%d"))
        self.desc_text.delete("1.0", tk.END)
        self.authors_entry.delete(0, tk.END)
        self.keywords_entry.delete(0, tk.END)

        # –°–±—Ä–∞—Å—ã–≤–∞–µ–º –∫–æ–º–ø–µ—Ç–µ–Ω—Ü–∏–∏
        for var, _ in self.competency_vars:
            var.set(False)
        for level_var in self.level_vars:
            level_var.set(3)

                level_text = f"{avg_level:.1f}/5.0"
